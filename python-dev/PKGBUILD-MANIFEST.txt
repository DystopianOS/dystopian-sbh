═══════════════════════════════════════════════════════════════════════════

Provides three core modules:

USAGE EXAMPLE
═══════════════════════════════════════════════════════════════════════════
#!/usr/bin/env python3
import asyncio
from cachyos_sbh import tpm_async, cryptsetup_async

async def main():
    # Seal TPM2 to Secure Boot state
    await tpm_async.tpm_seal_pcrs([7, 11], b"boot-hash")
    
    # Open LUKS devices
    success = await cryptsetup_async.cryptsetup_open(
        "/dev/nvme0n1p2", "luks1", "passphrase"
    )
    print(f"Device open: {success}")

asyncio.run(main())
